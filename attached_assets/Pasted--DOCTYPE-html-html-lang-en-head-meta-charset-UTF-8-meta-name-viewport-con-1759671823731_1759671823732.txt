            const wheelText = document.getElementById('wheel-text');
            
            isWheelExpanded = !isWheelExpanded;
            
            if (isWheelExpanded) {
                wheelCenter.classList.add('rotating');
                navItems.classList.add('expanded');
                navWheel.classList.add('expanded');
                wheelText.textContent = '‚úï';
                
                // Highlight first item
                updateHighlight(0);
                totalRotation = 0;
                navItems.style.transform = `translate(-50%, -50%) scale(1.1) rotate(0deg)`;
            } else {
                wheelCenter.classList.remove('rotating');
                navItems.classList.remove('expanded');
                navWheel.classList.remove('expanded');
                wheelText.textContent = 'MENU';
                
                // Clear highlights
                document.querySelectorAll('.nav-item').forEach(item => {
                    item.classList.remove('active-at-top');
                });
            }
        }

        function navigateTo(viewId, clickedItem, index) {
            document.querySelectorAll('.nav-item').forEach(item => {
                item.classList.remove('active-at-top');
            });
            
            clickedItem.classList.add('active-at-top');
            switchToView(viewId);
            currentIndex = index;
            
            setTimeout(() => {
                if (isWheelExpanded) {
                    toggleWheel();
                }
            }, 500);
        }

        function updateHighlight(index) {
            document.querySelectorAll('.nav-item').forEach(item => {
                item.classList.remove('active-at-top');
            });
            
            const items = document.querySelectorAll('.nav-item');
            if (items[index]) {
                items[index].classList.add('active-at-top');
                switchToView(views[index]);
                currentIndex = index;
            }
        }

        function getAngle(centerX, centerY, pointX, pointY) {
            return Math.atan2(pointY - centerY, pointX - centerX) * 180 / Math.PI;
        }

        function getCenter(element) {
            const rect = element.getBoundingClientRect();
            return {
                x: rect.left + rect.width / 2,
                y: rect.top + rect.height / 2
            };
        }

        // Drag functionality
        function handleStart(e) {
            if (!isWheelExpanded) return;
            
            isDragging = true;
            const navItems = document.getElementById('nav-items');
            const center = getCenter(navItems);
            const point = e.touches ? e.touches[0] : e;
            
            lastAngle = getAngle(center.x, center.y, point.clientX, point.clientY);
            navItems.style.transition = 'none';
            e.preventDefault();
        }

        function updateIconRotations() {
            const icons = document.querySelectorAll('.nav-item-icon');
            icons.forEach(icon => {
                icon.style.transform = `rotate(${-totalRotation}deg)`;
            });
        }

        function handleMove(e) {
            if (!isDragging || !isWheelExpanded) return;
            
            const navItems = document.getElementById('nav-items');
            const center = getCenter(navItems);
            const point = e.touches ? e.touches[0] : e;
            
            const currentAngle = getAngle(center.x, center.y, point.clientX, point.clientY);
            let angleDelta = currentAngle - lastAngle;
            
            // Normalize angle
            if (angleDelta > 180) angleDelta -= 360;
            if (angleDelta < -180) angleDelta += 360;
            
            totalRotation += angleDelta * 0.5; // Damping
            navItems.style.transform = `translate(-50%, -50%) scale(1.1) rotate(${totalRotation}deg)`;
            
            // Counter-rotate icons to keep them upright
            updateIconRotations();
            
            // Update highlight based on rotation
            const normalizedRotation = (((-totalRotation) % 360) + 360) % 360;
            const highlightIndex = Math.round(normalizedRotation / 36) % 10;
            updateHighlight(highlightIndex);
            
            lastAngle = currentAngle;
            e.preventDefault();
        }

        function handleEnd(e) {
            if (!isDragging) return;
            
            isDragging = false;
            const navItems = document.getElementById('nav-items');
            
            // Snap to nearest position
            const snapAngle = Math.round(totalRotation / 36) * 36;
            navItems.style.transition = 'transform 0.5s ease-out';
            totalRotation = snapAngle;
            navItems.style.transform = `translate(-50%, -50%) scale(1.1) rotate(${snapAngle}deg)`;
            
            // Update icon rotations with transition
            const icons = document.querySelectorAll('.nav-item-icon');
            icons.forEach(icon => {
                icon.style.transition = 'transform 0.5s ease-out';
                icon.style.transform = `rotate(${-snapAngle}deg)`;
            });
            
            setTimeout(() => {
                navItems.style.transition = 'none';
                icons.forEach(icon => {
                    icon.style.transition = 'transform 0.3s ease';
                });
            }, 500);
        }

        // Event listeners
        document.addEventListener('DOMContentLoaded', function() {
            const navItems = document.getElementById('nav-items');
            
            // Mouse events
            navItems.addEventListener('mousedown', handleStart);
            document.addEventListener('mousemove', handleMove);
            document.addEventListener('mouseup', handleEnd);
            
            // Touch events
            navItems.addEventListener('touchstart', handleStart, { passive: false });
            document.addEventListener('touchmove', handleMove, { passive: false });
            document.addEventListener('touchend', handleEnd);
            
            // Prevent context menu
            navItems.addEventListener('contextmenu', e => e.preventDefault());
        });
    </script>
