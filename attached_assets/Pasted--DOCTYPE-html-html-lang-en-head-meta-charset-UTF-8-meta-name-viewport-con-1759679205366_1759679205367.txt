    <script>
        // Handle slider
        const slider = document.querySelector('.erocks-slider');
        const sliderContainer = document.querySelector('.erocks-slider-container');
        const sliderButton = document.querySelector('.erocks-slider-button');
        const sliderImg = sliderButton.querySelector('img');
        const sliderSteps = document.querySelectorAll('.erocks-slider-step');
        let snapTimeout;
        let isJumping = false;
        let isDragging = false;

        function updateSliderPosition(jumping = false) {
            const value = parseFloat(slider.value);
            const percentage = value / 4;
            const maxTravel = 200 - 36;
            const leftPosition = 2 + (percentage * maxTravel);
            
            // Handle jumping animation
            if (jumping && !isDragging) {
                sliderButton.classList.add('jumping');
                isJumping = true;
                setTimeout(() => {
                    sliderButton.classList.remove('jumping');
                    isJumping = false;
                }, 200);
            }
            
            // Update button position
            sliderButton.style.left = leftPosition + 'px';
            
            // Update rotation
            const rotation = value * 360;
            sliderImg.style.transform = `rotate(${rotation}deg)`;
            
            // Handle snap-to-upright effect
            clearTimeout(snapTimeout);
            sliderButton.classList.remove('snapping');
            
            if (!isDragging) {
                snapTimeout = setTimeout(() => {
                    const nearestStep = Math.round(value);
                    const snapRotation = nearestStep * 360;
                    sliderButton.classList.add('snapping');
                    sliderImg.style.transform = `rotate(${snapRotation}deg)`;
                }, 250);
            }
        }

        function handleActiveStart() {
            isDragging = true;
            sliderContainer.classList.add('active');
            sliderButton.classList.remove('jumping');
        }

        function handleActiveEnd() {
            isDragging = false;
            sliderContainer.classList.remove('active');
            // Ensure snap happens after drag ends
            updateSliderPosition(false);
        }

        // Main slider event listeners
        slider.addEventListener('input', () => {
            if (isDragging) {
                updateSliderPosition(false);
            }
        });

        // Mouse events
        slider.addEventListener('mousedown', handleActiveStart);
        slider.addEventListener('mouseup', handleActiveEnd);
        
        // Handle mouse leave during drag
        slider.addEventListener('mouseleave', (e) => {
            if (isDragging) {
                handleActiveEnd();
            }
        });
        
        // Touch events
        slider.addEventListener('touchstart', handleActiveStart, { passive: true });
        slider.addEventListener('touchend', handleActiveEnd, { passive: true });
        slider.addEventListener('touchcancel', handleActiveEnd, { passive: true });
        
        // Click on step indicators to jump
        sliderSteps.forEach(step => {
            step.addEventListener('click', (e) => {
                e.stopPropagation();
                if (isDragging) return;
                
                const targetValue = parseFloat(step.dataset.value);
                slider.value = targetValue;
                updateSliderPosition(true);
                
                // Flash active state
                handleActiveStart();
                setTimeout(() => {
                    isDragging = false;
                    sliderContainer.classList.remove('active');
                }, 100);
            });
        });

        // Click on track to jump
        sliderContainer.addEventListener('click', (e) => {
            // Ignore clicks on slider itself, during jumps, or during drags
            if (e.target === slider || isJumping || isDragging) return;
            
            const rect = sliderContainer.getBoundingClientRect();
            const x = e.clientX - rect.left;
            const percentage = Math.max(0, Math.min(1, (x - 18) / (200 - 36)));
            const newValue = percentage * 4;
            
            slider.value = newValue;
            updateSliderPosition(true);
            
            // Flash active state
            handleActiveStart();
            setTimeout(() => {
                isDragging = false;
                sliderContainer.classList.remove('active');
            }, 100);
        });
        
        // Initialize
        updateSliderPosition(false);
    </script>
